int sensorstate(double value, double sd)
{
    if( value < 2*sd)
    {
        sensorcond = 0;
        return sensorcond;
    }
    else if(value > 2*sd && value < 3*sd )
    {
        sensorcond =1;
        return sensorcond;
    }
    else if(value >3*sd)
    {
        sensorcond =2;
        return sensorcond;
    }
}
